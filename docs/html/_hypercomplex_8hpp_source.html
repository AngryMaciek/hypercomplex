296// calculate norm of the number
297template <typename T, const unsigned int dim>
298inline T Hypercomplex<T, dim>::norm() const {
299 T result = T();
300 for (unsigned int i=0; i < dim; i++) result += arr[i] * arr[i];
301 return sqrt(result);
302}

304// calculate inverse of the number
305template <typename T, const unsigned int dim>
306Hypercomplex<T, dim> Hypercomplex<T, dim>::inv() const {
307 T zero = T();
308 T norm = (*this).norm();
309 if (norm == zero) {
310 throw std::invalid_argument("division by zero");
311 } else {
312 T temparr[dim]; // NOLINT
313 temparr[0] = arr[0] / (norm * norm);
314 for (unsigned int i=1; i < dim; i++)
315 temparr[i] = -arr[i] / (norm * norm);
316 Hypercomplex<T, dim> H(temparr);
317 return H;
318 }
319}

424// overloaded * binary operator
425template <typename T, const unsigned int dim>
426Hypercomplex<T, dim> operator*(
427 const Hypercomplex<T, dim> &H1,
428 const Hypercomplex<T, dim> &H2
429) {
430 // recursion base:
431 if constexpr (dim == 1) {
432 T temparr[] = { H1[0] * H2[0] };
433 Hypercomplex<T, 1> H_(temparr);
434 return H_;
435 // recursion step:
436 } else {
437 // shared objects:
438 const unsigned int halfd = dim / 2;
439 T temparr[dim]; // NOLINT
440 // construct helper objects:
441 for (unsigned int i=0; i < halfd; i++) temparr[i] = H1[i];
442 Hypercomplex<T, halfd> H1a(temparr);
443 for (unsigned int i=0; i < halfd; i++) temparr[i] = H1[i+halfd];
444 Hypercomplex<T, halfd> H1b(temparr);
445 for (unsigned int i=0; i < halfd; i++) temparr[i] = H2[i];
446 Hypercomplex<T, halfd> H2a(temparr);
447 for (unsigned int i=0; i < halfd; i++) temparr[i] = H2[i+halfd];
448 Hypercomplex<T, halfd> H2b(temparr);
449 // multiply recursively:
450 Hypercomplex<T, halfd> H1a2a = H1a * H2a;
451 Hypercomplex<T, halfd> H2b_1b = ~H2b * H1b;
452 Hypercomplex<T, halfd> H2b1a = H2b * H1a;
453 Hypercomplex<T, halfd> H1b2a_ = H1b * ~H2a;
454 // construct the final object
455 Hypercomplex<T, halfd> Ha = H1a2a - H2b_1b;
456 Hypercomplex<T, halfd> Hb = H2b1a + H1b2a_;
457 for (unsigned int i=0; i < halfd; i++) temparr[i] = Ha[i];
458 for (unsigned int i=0; i < halfd; i++) temparr[i+halfd] = Hb[i];
459 Hypercomplex<T, dim> H(temparr);
460 return H;
461 }
462}

564// calculate e^H
565template <typename T, const unsigned int dim>
566Hypercomplex<T, dim> exp(const Hypercomplex<T, dim> &H) {
567 Hypercomplex<T, dim> result = Im(H);
568 T zero = T();
569 T norm = result.norm();
570 if (norm == zero) {
571 result[0] = exp(H[0]);
572 for (unsigned int i=1; i < dim; i++) result[i] = zero;
573 } else {
574 T sinv_v = sin(norm) / norm;
575 for (unsigned int i=0; i < dim; i++) result[i] *= sinv_v;
576 result[0] += cos(norm);
577 for (unsigned int i=0; i < dim; i++) result[i] *= exp(H[0]);
578 }
579 return result;
580}

629 if ((dim & (dim - 1)) != 0) {
630 throw std::invalid_argument("invalid dimension");
631 }
632 for (unsigned int i=0; i < dim; i++)
633 mpfr_init2(arr[i], MPFR_global_precision);

648 mpfr_set(arr[i], H[i], MPFR_RNDN);

661
671 int norm(mpfr_t norm) const {
672 mpfr_t temp;
673 mpfr_init2(temp, MPFR_global_precision);
674 mpfr_set_zero(norm, 0);
675 for (unsigned int i=0; i < dim; i++) {
676 mpfr_mul(temp, arr[i], arr[i], MPFR_RNDN);
677 mpfr_add(norm, norm, temp, MPFR_RNDN);
678 }
679 mpfr_sqrt(norm, norm, MPFR_RNDN);
680 mpfr_clear(temp);
681 return 0;
682 }
683
687 Hypercomplex inv() const {
688 mpfr_t zero, norm;
689 mpfr_init2(zero, MPFR_global_precision);
690 mpfr_init2(norm, MPFR_global_precision);
691 mpfr_set_zero(zero, 0);
692 (*this).norm(norm);
693 if (mpfr_equal_p(norm, zero)) {
694 mpfr_clear(zero);
695 mpfr_clear(norm);
696 throw std::invalid_argument("division by zero");
697 } else {
698 mpfr_t temparr[dim]; // NOLINT
699 for (unsigned int i=0; i < dim; i++)
700 mpfr_init2(temparr[i], MPFR_global_precision);
701 mpfr_mul(norm, norm, norm, MPFR_RNDN);
702 mpfr_div(temparr[0], arr[0], norm, MPFR_RNDN);
703 for (unsigned int i=1; i < dim; i++) {
704 mpfr_div(temparr[i], arr[i], norm, MPFR_RNDN);
705 mpfr_sub(temparr[i], zero, temparr[i], MPFR_RNDN);
706 }
707 Hypercomplex<mpfr_t, dim> H(temparr);
708 mpfr_clear(zero);
709 mpfr_clear(norm);
710 for (unsigned int i=0; i < dim; i++) mpfr_clear(temparr[i]);
711 return H;
712 }
713 }

940 mpfr_mul(result, H1[0], H2[0], MPFR_RNDN);

1068template <const unsigned int dim>
1069Hypercomplex<mpfr_t, dim> exp(const Hypercomplex<mpfr_t, dim> &H) {
1070 Hypercomplex<mpfr_t, dim> result = Im(H);
1071 mpfr_t zero, norm, expreal;
1072 mpfr_init2(zero, MPFR_global_precision);
1073 mpfr_init2(norm, MPFR_global_precision);
1074 mpfr_init2(expreal, MPFR_global_precision);
1075 mpfr_set_zero(zero, 0);
1076 result.norm(norm);
1077 mpfr_exp(expreal, H[0], MPFR_RNDN);
1078
1079 if (mpfr_equal_p(norm, zero)) {
1080 mpfr_set(result[0], expreal, MPFR_RNDN);
1081 for (unsigned int i=1; i < dim; i++) mpfr_set_zero(result[i], 0);
1082 } else {
1083 mpfr_t sinv_v;
1084 mpfr_init2(sinv_v, MPFR_global_precision);
1085 mpfr_sin(sinv_v, norm, MPFR_RNDN);
1086 mpfr_div(sinv_v, sinv_v, norm, MPFR_RNDN);
1087 for (unsigned int i=0; i < dim; i++) {
1088 mpfr_mul(result[i], result[i], sinv_v, MPFR_RNDN);
1089 }
1090 mpfr_cos(norm, norm, MPFR_RNDN);
1091 mpfr_add(result[0], result[0], norm, MPFR_RNDN);
1092 for (unsigned int i=0; i < dim; i++) {
1093 mpfr_mul(result[i], result[i], expreal, MPFR_RNDN);
1094 }
1095 mpfr_clear(sinv_v);
1096 }
1097 mpfr_clear(zero);
1098 mpfr_clear(norm);
1099 mpfr_clear(expreal);
1100 return result;
1101}
