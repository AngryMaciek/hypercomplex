38template <typename T, const unsigned int dim>
39class Hypercomplex {
40 private:
41 T arr[dim]; // NOLINT
42 static inline uint64_t** baseprodabs;
43 static inline bool** baseprodpos;
44
45 public:
48 static void init() {
49 int64_t** M = new int64_t*[dim];
50 for (unsigned int i = 0; i < dim; i++) M[i] = new int64_t[dim];
51 M[0][0] = 1;
52 unsigned int n = 1;
53 while (n != dim) {
54 for (unsigned int i=0; i < n; i++) {
55 for (unsigned int j=0; j < n; j++) {
56 M[i][n+j] = M[j][i] > 0 ? M[j][i] + n : M[j][i] - n;
57 M[i+n][j] = M[i][j] > 0 ? M[i][j] + n : M[i][j] - n;
58 M[i+n][j] = M[i+n][j] * (j ? -1 : 1);
59 M[i+n][j+n] = -M[j][i] * (j ? -1 : 1);
60 }
61 }
62 n *= 2;
63 }
64 baseprodabs = new uint64_t*[dim];
65 baseprodpos = new bool*[dim];
66 for (unsigned int i = 0; i < dim; i++) {
67 baseprodabs[i] = new uint64_t[dim];
68 baseprodpos[i] = new bool[dim];
69 }
70 for (unsigned int i=0; i < dim; i++) {
71 for (unsigned int j=0; j < dim; j++) {
72 baseprodabs[i][j] = std::abs(M[i][j]) - 1;
73 baseprodpos[i][j] = (0 < M[i][j]);
74 }
75 }
76 for (unsigned int i = 0; i < dim; i++) {
77 delete[] M[i];
78 }
79 delete[] M;
80 }

84 static void clear() {
85 for (unsigned int i = 0; i < dim; i++) {
86 delete[] baseprodabs[i];
87 delete[] baseprodpos[i];
88 }
89 delete[] baseprodabs;

91 }

116 return H;

376// calculate inverse of the number
377template <typename T, const unsigned int dim>
378Hypercomplex<T, dim> Hypercomplex<T, dim>::inv() const {
379 T zero = T();
380 T norm = (*this).norm();
381 if (norm == zero) {
382 throw std::invalid_argument("division by zero");
383 } else {
384 T temparr[dim]; // NOLINT
385 temparr[0] = arr[0] / (norm * norm);
386 for (unsigned int i=1; i < dim; i++)
387 temparr[i] = -arr[i] / (norm * norm);
388 Hypercomplex<T, dim> H(temparr);
389 return H;
390 }

496// overloaded * binary operator
497template <typename T, const unsigned int dim>
498Hypercomplex<T, dim> operator*(
499 const Hypercomplex<T, dim> &H1,
500 const Hypercomplex<T, dim> &H2
501) {
502 // recursion base:
503 if constexpr (dim == 1) {
504 T temparr[] = { H1[0] * H2[0] };
505 Hypercomplex<T, 1> H_(temparr);
506 return H_;
507 // recursion step:
508 } else {
509 // shared objects:
510 const unsigned int halfd = dim / 2;
511 T temparr[dim]; // NOLINT
512 // construct helper objects:
513 for (unsigned int i=0; i < halfd; i++) temparr[i] = H1[i];
514 Hypercomplex<T, halfd> H1a(temparr);
515 for (unsigned int i=0; i < halfd; i++) temparr[i] = H1[i+halfd];
516 Hypercomplex<T, halfd> H1b(temparr);
517 for (unsigned int i=0; i < halfd; i++) temparr[i] = H2[i];
518 Hypercomplex<T, halfd> H2a(temparr);
519 for (unsigned int i=0; i < halfd; i++) temparr[i] = H2[i+halfd];
520 Hypercomplex<T, halfd> H2b(temparr);
521 // multiply recursively:
522 Hypercomplex<T, halfd> H1a2a = H1a * H2a;
523 Hypercomplex<T, halfd> H2b_1b = ~H2b * H1b;
524 Hypercomplex<T, halfd> H2b1a = H2b * H1a;
525 Hypercomplex<T, halfd> H1b2a_ = H1b * ~H2a;
526 // construct the final object
527 Hypercomplex<T, halfd> Ha = H1a2a - H2b_1b;
528 Hypercomplex<T, halfd> Hb = H2b1a + H1b2a_;
529 for (unsigned int i=0; i < halfd; i++) temparr[i] = Ha[i];
530 for (unsigned int i=0; i < halfd; i++) temparr[i+halfd] = Hb[i];
531 Hypercomplex<T, dim> H(temparr);
532 return H;
533 }
534}

636// calculate e^H
637template <typename T, const unsigned int dim>
638Hypercomplex<T, dim> exp(const Hypercomplex<T, dim> &H) {
639 Hypercomplex<T, dim> result = Im(H);
640 T zero = T();
641 T norm = result.norm();
642 if (norm == zero) {
643 result[0] = exp(H[0]);
644 for (unsigned int i=1; i < dim; i++) result[i] = zero;
645 } else {
646 T sinv_v = sin(norm) / norm;
647 for (unsigned int i=0; i < dim; i++) result[i] *= sinv_v;
648 result[0] += cos(norm);
649 for (unsigned int i=0; i < dim; i++) result[i] *= exp(H[0]);
650 }
651 return result;
652}

748 const Hypercomplex &H1,
749 const Hypercomplex &H2
750 ) {
751 mpfr_t prod;
752 mpfr_init2(prod, MPFR_global_precision);
753 mpfr_t temparr[dim]; // NOLINT

776 for (unsigned int i=0; i < dim; i++) mpfr_clear(temparr[i]);
777 mpfr_clear(prod);

849 mpfr_init2(zero, MPFR_global_precision);
850 mpfr_init2(norm, MPFR_global_precision);
851 mpfr_set_zero(zero, 0);

868 mpfr_clear(zero);

1091template <const unsigned int dim>
1092Hypercomplex<mpfr_t, dim> operator*(
1093 const Hypercomplex<mpfr_t, dim> &H1,
1094 const Hypercomplex<mpfr_t, dim> &H2
1095) {
1096 // recursion base:
1097 if constexpr (dim == 1) {
1098 mpfr_t result;
1099 mpfr_init2(result, MPFR_global_precision);
1100 mpfr_mul(result, H1[0], H2[0], MPFR_RNDN);
1101 mpfr_t temparr[1];
1102 mpfr_init2(temparr[0], MPFR_global_precision);
1103 mpfr_set(temparr[0], result, MPFR_RNDN);
1104 Hypercomplex<mpfr_t, 1> H_(temparr);
1105 mpfr_clear(result);
1106 mpfr_clear(temparr[0]);
1107 return H_;
1108 // recursion step:
1109 } else {
1110 // shared objects:
1111 const unsigned int halfd = dim / 2;
1112 mpfr_t temparr[dim]; // NOLINT
1113 for (unsigned int i=0; i < dim; i++)
1114 mpfr_init2(temparr[i], MPFR_global_precision);
1115 // construct helper objects:
1116 for (unsigned int i=0; i < halfd; i++)
1117 mpfr_set(temparr[i], H1[i], MPFR_RNDN);
1118 Hypercomplex<mpfr_t, halfd> H1a(temparr);
1119 for (unsigned int i=0; i < halfd; i++)
1120 mpfr_set(temparr[i], H1[i+halfd], MPFR_RNDN);
1121 Hypercomplex<mpfr_t, halfd> H1b(temparr);
1122 for (unsigned int i=0; i < halfd; i++)
1123 mpfr_set(temparr[i], H2[i], MPFR_RNDN);
1124 Hypercomplex<mpfr_t, halfd> H2a(temparr);
1125 for (unsigned int i=0; i < halfd; i++)
1126 mpfr_set(temparr[i], H2[i+halfd], MPFR_RNDN);
1127 Hypercomplex<mpfr_t, halfd> H2b(temparr);
1128 // multiply recursively:
1129 Hypercomplex<mpfr_t, halfd> H1a2a = H1a * H2a;
1130 Hypercomplex<mpfr_t, halfd> H2b_1b = ~H2b * H1b;
1131 Hypercomplex<mpfr_t, halfd> H2b1a = H2b * H1a;
1132 Hypercomplex<mpfr_t, halfd> H1b2a_ = H1b * ~H2a;
1133 // construct the final object
1134 Hypercomplex<mpfr_t, halfd> Ha = H1a2a - H2b_1b;
1135 Hypercomplex<mpfr_t, halfd> Hb = H2b1a + H1b2a_;
1136 for (unsigned int i=0; i < halfd; i++)
1137 mpfr_set(temparr[i], Ha[i], MPFR_RNDN);
1138 for (unsigned int i=0; i < halfd; i++)
1139 mpfr_set(temparr[i+halfd], Hb[i], MPFR_RNDN);
1140 Hypercomplex<mpfr_t, dim> H(temparr);
1141 for (unsigned int i=0; i < dim; i++) mpfr_clear(temparr[i]);
1142 return H;
1143 }
1144}

1233 mpfr_init2(norm, MPFR_global_precision);
1234 mpfr_init2(expreal, MPFR_global_precision);

1237 mpfr_exp(expreal, H[0], MPFR_RNDN);

1240 mpfr_set(result[0], expreal, MPFR_RNDN);
1241 for (unsigned int i=1; i < dim; i++) mpfr_set_zero(result[i], 0);

1245 mpfr_sin(sinv_v, norm, MPFR_RNDN);
1246 mpfr_div(sinv_v, sinv_v, norm, MPFR_RNDN);
1247 for (unsigned int i=0; i < dim; i++) {
1248 mpfr_mul(result[i], result[i], sinv_v, MPFR_RNDN);
1249 }
1250 mpfr_cos(norm, norm, MPFR_RNDN);
1251 mpfr_add(result[0], result[0], norm, MPFR_RNDN);

1253 mpfr_mul(result[i], result[i], expreal, MPFR_RNDN);
